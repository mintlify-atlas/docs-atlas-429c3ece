---
title: Link Component
description: Enable fast client-side navigation with the built-in next/link component, which supports prefetching and client-side transitions.
---

`<Link>` is a React component that extends the HTML `<a>` element to provide [prefetching](/app/getting-started/linking-and-navigating#prefetching) and client-side navigation between routes. It is the primary way to navigate between routes in Next.js.

```tsx filename="app/page.tsx"
import Link from 'next/link'

export default function Page() {
  return <Link href="/dashboard">Dashboard</Link>
}
```

## Props

| Prop | Example | Type | Required |
| --- | --- | --- | --- |
| [`href`](#href) | `href="/dashboard"` | String or Object | Yes |
| [`replace`](#replace) | `replace={false}` | Boolean | — |
| [`scroll`](#scroll) | `scroll={false}` | Boolean | — |
| [`prefetch`](#prefetch) | `prefetch={false}` | Boolean or null | — |
| [`onNavigate`](#onnavigate) | `onNavigate={(e) => {}}` | Function | — |
| [`transitionTypes`](#transitiontypes) | `transitionTypes={['slide-in']}` | `string[]` | — |

<Note>`<a>` tag attributes such as `className` or `target="_blank"` can be added to `<Link>` as props and will be passed to the underlying `<a>` element.</Note>

<ParamField path="href" type="string | object" required>
  The path or URL to navigate to.

  ```tsx filename="app/page.tsx"
  import Link from 'next/link'

  // Navigate to /about?name=test
  export default function Page() {
    return (
      <Link
        href={{
          pathname: '/about',
          query: { name: 'test' },
        }}
      >
        About
      </Link>
    )
  }
  ```
</ParamField>

<ParamField path="replace" type="boolean">
  Defaults to `false`. When `true`, replaces the current history state instead of pushing a new URL into the browser history stack.
</ParamField>

<ParamField path="scroll" type="boolean">
  Defaults to `true`. The default behavior maintains scroll position when navigating to a new page if it's visible in the viewport. When not visible, Next.js scrolls to the top of the first page element.

  Setting `scroll={false}` disables this behavior.

  ```tsx filename="app/page.tsx"
  import Link from 'next/link'

  export default function Page() {
    return (
      <Link href="/dashboard" scroll={false}>
        Dashboard
      </Link>
    )
  }
  ```
</ParamField>

<ParamField path="prefetch" type="boolean | null">
  Controls prefetching behavior. Prefetching only occurs in production.

  - **`"auto"` or `null` (default)**: For static routes, prefetches the full route. For dynamic routes, prefetches down to the nearest `loading.js` boundary.
  - `true`: Prefetches the full route for both static and dynamic routes.
  - `false`: Disables prefetching on viewport entry and hover.
</ParamField>

<ParamField path="onNavigate" type="function">
  An event handler called during client-side navigation. The handler receives an event object with a `preventDefault()` method to cancel navigation.

  ```tsx filename="app/page.tsx"
  import Link from 'next/link'

  export default function Page() {
    return (
      <Link
        href="/dashboard"
        onNavigate={(e) => {
          console.log('Navigating...')
          // e.preventDefault() to cancel
        }}
      >
        Dashboard
      </Link>
    )
  }
  ```

  <Note>`onClick` fires for all clicks; `onNavigate` only fires during client-side navigation. Modifier key clicks, external URLs, and `download` links do not trigger `onNavigate`.</Note>
</ParamField>

<ParamField path="transitionTypes" type="string[]">
  A list of transition types passed to `React.addTransitionType` inside the navigation transition, enabling `<ViewTransition>` components to apply different animations.

  ```tsx filename="app/page.tsx"
  import Link from 'next/link'

  export default function Page() {
    return (
      <Link href="/about" transitionTypes={['slide-in']}>
        About
      </Link>
    )
  }
  ```
</ParamField>

## Examples

### Linking to dynamic route segments

```tsx filename="app/blog/post-list.tsx"
import Link from 'next/link'

interface Post {
  id: number
  title: string
  slug: string
}

export default function PostList({ posts }: { posts: Post[] }) {
  return (
    <ul>
      {posts.map((post) => (
        <li key={post.id}>
          <Link href={`/blog/${post.slug}`}>{post.title}</Link>
        </li>
      ))}
    </ul>
  )
}
```

### Checking active links

```tsx filename="app/ui/nav-links.tsx"
'use client'

import { usePathname } from 'next/navigation'
import Link from 'next/link'

export function Links() {
  const pathname = usePathname()

  return (
    <nav>
      <Link className={`link ${pathname === '/' ? 'active' : ''}`} href="/">
        Home
      </Link>
      <Link
        className={`link ${pathname === '/about' ? 'active' : ''}`}
        href="/about"
      >
        About
      </Link>
    </nav>
  )
}
```

### Scrolling to an id

```jsx
<Link href="/dashboard#settings">Settings</Link>

// Output
<a href="/dashboard#settings">Settings</a>
```

### Disable scrolling to the top

```tsx filename="app/page.tsx"
import Link from 'next/link'

export default function Page() {
  return (
    <Link href="/#hashid" scroll={false}>
      Disables scrolling to the top
    </Link>
  )
}
```

### Blocking navigation

Use `onNavigate` to block navigation when a form has unsaved changes:

```tsx filename="app/components/custom-link.tsx"
'use client'

import Link from 'next/link'
import { useNavigationBlocker } from '../contexts/navigation-blocker'

export function CustomLink({ children, ...props }) {
  const { isBlocked } = useNavigationBlocker()

  return (
    <Link
      onNavigate={(e) => {
        if (
          isBlocked &&
          !window.confirm('You have unsaved changes. Leave anyway?')
        ) {
          e.preventDefault()
        }
      }}
      {...props}
    >
      {children}
    </Link>
  )
}
```

### Scroll offset with sticky headers

Use `scroll-padding-top` to account for sticky header height:

```css filename="app/globals.css"
html {
  scroll-padding-top: 64px; /* Match the height of your sticky header */
}
```
