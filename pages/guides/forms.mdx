Forms enable you to create and update data in web applications. In the Pages Router, form submissions are handled by sending requests to API Routes.

## Server-side form handling

Create an API Route to handle form submissions:

<Tabs>
  <Tab title="TypeScript">
  ```ts filename="pages/api/submit.ts"
  import type { NextApiRequest, NextApiResponse } from 'next'

  export default async function handler(
    req: NextApiRequest,
    res: NextApiResponse
  ) {
    const data = req.body
    const id = await createItem(data)
    res.status(200).json({ id })
  }
  ```
  </Tab>
  <Tab title="JavaScript">
  ```js filename="pages/api/submit.js"

    const data = req.body
    res.status(200).json({ data })
  }
  ```
  </Tab>
</Tabs>

Call the API Route from the client with an event handler:

<Tabs>
  <Tab title="TypeScript">
  ```tsx filename="pages/index.tsx"
  import { FormEvent } from 'react'

  export default function Page() {
    async function onSubmit(event: FormEvent<HTMLFormElement>) {
      event.preventDefault()

      const formData = new FormData(event.currentTarget)
      const response = await fetch('/api/submit', {
        method: 'POST',
        body: formData,
      })

      const data = await response.json()
      // ...
    }

    return (
      <form onSubmit={onSubmit}>
        <input type="text" name="name" />
        <button type="submit">Submit</button>
      </form>
    )
  }
  ```

<Note>
  API Routes run on the server, so you can safely use environment variables (like API keys) without exposing them to the client.
</Note>

## Form validation

Use HTML validation attributes for basic client-side validation:

```html
<input type="email" name="email" required />
<input type="text" name="name" minlength="2" required />
```

For server-side validation, use a schema validation library like [Zod](https://zod.dev/):

```ts filename="pages/api/submit.ts"
import type { NextApiRequest, NextApiResponse } from 'next'
import { z } from 'zod'

const schema = z.object({
  name: z.string().min(2),
  email: z.string().email(),
})

export default async function handler(
  req: NextApiRequest,
  res: NextApiResponse
) {
  const parsed = schema.safeParse(req.body)
  if (!parsed.success) {
    return res.status(400).json({ error: parsed.error })
  }
  // ...
}
```

## Error handling

Display error messages when form submission fails:

```tsx filename="pages/index.tsx"
import React, { useState, FormEvent } from 'react'

export default function Page() {
  const [isLoading, setIsLoading] = useState<boolean>(false)
  const [error, setError] = useState<string | null>(null)

  async function onSubmit(event: FormEvent<HTMLFormElement>) {
    event.preventDefault()
    setIsLoading(true)
    setError(null)

    try {
      const formData = new FormData(event.currentTarget)
      const response = await fetch('/api/submit', {
        method: 'POST',
        body: formData,
      })

      if (!response.ok) {
        throw new Error('Failed to submit the data. Please try again.')
      }

      const data = await response.json()
      // ...
    } catch (error) {
      setError(error.message)
      console.error(error)
    } finally {
      setIsLoading(false)
    }
  }

  return (
    <div>
      {error && <div style={{ color: 'red' }}>{error}</div>}
      <form onSubmit={onSubmit}>
        <input type="text" name="name" required />
        <button type="submit" disabled={isLoading}>
          {isLoading ? 'Loading...' : 'Submit'}
        </button>
      </form>
    </div>
  )
}
```

## Redirecting after submission

Redirect the user after a successful form submission:

```ts filename="pages/api/submit.ts"
import type { NextApiRequest, NextApiResponse } from 'next'

export default async function handler(
  req: NextApiRequest,
  res: NextApiResponse
) {
  const id = await addPost()
  res.redirect(307, `/post/${id}`)
}
```

## Optimistic updates

For a better user experience, you can update the UI optimistically before the server confirms the mutation:

```tsx
import { useState } from 'react'

export default function Page() {
  const [items, setItems] = useState(initialItems)

  async function onSubmit(event) {
    event.preventDefault()
    const newItem = { id: Date.now(), name: event.target.name.value }

    // Optimistically update the UI
    setItems([...items, newItem])

    try {
      await fetch('/api/items', {
        method: 'POST',
        body: JSON.stringify(newItem),
      })
    } catch (error) {
      // Revert the optimistic update on failure
      setItems(items)
    }
  }

  return (
    <form onSubmit={onSubmit}>
      <input type="text" name="name" />
      <button type="submit">Add Item</button>
    </form>
  )
}
```
